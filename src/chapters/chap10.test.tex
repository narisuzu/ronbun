\subsubsection{获取车站布局测试}
测试输入（其中的实例id是之前创建的）：
\begin{lstlisting}
query {
  stationLayout(id: "db42f0ed-d096-4805-b048-909e69dd44e2") {
    title,
    nodes{
      nodeId, 
      trackId, 
      leftP{x,y}, 
      rightP{x,y}, 
      leftJoint, 
      rightJoint
    },
    signals{
      signalId,
      sgnType,
      sgnMnt,
      protectNodeId,
      side,
      dir,
      pos{x,y},
      btns
    }
  }
}
\end{lstlisting}
